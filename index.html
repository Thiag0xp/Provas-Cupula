'block' : 'none'; } document.getElementById('tel').oninput = e => { let v = e.target.value.replace(/\D/g, ''); e.target.value = v.length > 4 ? v.slice(0,4)+'-'+v.slice(4,8) : v; }; function logar() { if(document.getElementById('pass').value === SENHA) { isAdmin = true; document.getElementById('loginOverlay').style.display = 'none'; document.getElementById('adminPanel').style.display = 'block'; render(); } else { alert("Acesso Negado"); } } // Função auxiliar para converter arquivos para Base64 const toBase64 = file => new Promise((resolve, reject) => { const reader = new FileReader(); reader.readAsDataURL(file); reader.onload = () => resolve(reader.result); reader.onerror = error => reject(error); }); async function salvar() { const db = JSON.parse(localStorage.getItem('cupula_final') || '{"membros":[],"provas":[]}'); const tipo = document.getElementById('tipo').value; if(tipo === 'membro') { const file = document.getElementById('fotoFile').files[0]; const foto = file ? await toBase64(file) : ''; db.membros.push({ uid: Date.now(), nome: document.getElementById('nome').value, id: document.getElementById('idMem').value, tel: document.getElementById('tel').value, foto, obs: document.getElementById('obs').value }); } else { const file = document.getElementById('fotoFile').files[0]; // Reaproveitando o campo de arquivo para Provas let mediaUrl = document.getElementById('videoLink').value; let fileType = 'link'; if(file) { mediaUrl = await toBase64(file); fileType = file.type.includes('video') ? 'video' : 'image'; } db.provas.push({ uid: Date.now(), titulo: document.getElementById('tituloP').value, link: mediaUrl, tipoMedia: fileType, obs: document.getElementById('obs').value }); } localStorage.setItem('cupula_final', JSON.stringify(db)); render(); alert("Dados sincronizados com sucesso."); } function apagar(cat, id) { if(!confirm("Apagar registro permanentemente?")) return; let db = JSON.parse(localStorage.getItem('cupula_final')); db[cat] = db[cat].filter(i => i.uid !== id); localStorage.setItem('cupula_final', JSON.stringify(db)); render(); } function render() { const db = JSON.parse(localStorage.getItem('cupula_final') || '{"membros":[],"provas":[]}'); const mList = document.getElementById('listaMembros'); mList.innerHTML = ''; db.membros.forEach(m => { mList.innerHTML += `
